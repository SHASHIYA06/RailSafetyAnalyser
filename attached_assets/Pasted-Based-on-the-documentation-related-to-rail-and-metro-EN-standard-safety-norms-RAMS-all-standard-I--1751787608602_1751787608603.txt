from flask import Flask, request, jsonify
from flask_sqlalchemy import SQLAlchemy
import os

app = Flask(__name__)
app.config['SQLALCHEMY_DATABASE_URI'] = os.getenv('DATABASE_URL', 'sqlite:///rams.db')
db = SQLAlchemy(app)

# Models
class Standard(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    code = db.Column(db.String, unique=True)
    title = db.Column(db.String)

class Clause(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    standard_code = db.Column(db.String, db.ForeignKey('standard.code'))
    ref = db.Column(db.String)
    content = db.Column(db.Text)

class Component(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    name = db.Column(db.String, unique=True)

class Compliance(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    comp_id = db.Column(db.Integer, db.ForeignKey('component.id'))
    clause_id = db.Column(db.Integer, db.ForeignKey('clause.id'))
    rams = db.Column(db.String)

    component = db.relationship('Component', backref='compliances')
    clause = db.relationship('Clause')

# Routes
@app.route('/api/search')
def search():
    q = request.args.get('q', '').lower()
    comps = Component.query.filter(Component.name.ilike(f'%{q}%')).all()
    res = []
    for c in comps:
        entries = []
        for cp in c.compliances:
            cl = cp.clause
            entries.append({
                'standard': cl.standard_code,
                'clause_ref': cl.ref,
                'snippet': cl.content[:200] + '…',
                'rams': cp.rams
            })
        res.append({'component': c.name, 'clauses': entries})
    return jsonify(res)

if __name__ == '__main__':
    db.create_all()
    # Example insert if empty

        s = Standard(code='EN50126', title='RAMS Lifecycle Standard')

            content='The Hazard Log shall include all hazards throughout the system life cycle.') )
        c = Component(name='Wheelset Bearing')
        db.session.add(c)
    db.session.commit()
    app.run(debug=True)

async function search() {
  const q = document.getElementById('search').value.trim();
  const res = await fetch(`/api/search?q=${encodeURIComponent(q)}`);
  const data = await res.json();
  const sec = document.getElementById('results');
  sec.innerHTML = '';
  if (!data.length) {
    sec.textContent = 'No results';
    return;
  }
  for (const comp of data) {
    const div = document.createElement('div');
    div.innerHTML = `<h3>${comp.component}</h3>`;

      div.appendChild(p);
    });
    sec.appendChild(div);
  }
}

    const li = document.createElement('li');
    li.innerHTML = `<a href="${s.pdf_link}" target="_blank">${s.code}: ${s.title}</a>`;
    ul.appendChild(li);
  });
}

## Features

- **Component Search**: Enter any electrical/mechanical/electronic component, instantly see all mapped EN standards, RAMS clauses, and compliance scores.

- **RAMS Tags**: Each clause is annotated with RAMS category (Safety, Reliability, etc.) and a compliance score.
- **Simple, Attractive UI**: Railway-themed, responsive, and easy to use.
- **Easy Setup**: Local SQLite DB, Flask backend, vanilla JS frontend.
